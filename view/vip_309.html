两个栈实现一个队列（超级详细）

队列实现“推小车”扑克牌游戏 用两个队列实现一个栈
上一节 下一节
栈和队列都属于线性存储结构，用来存储逻辑关系为“一对一”的数据。它们最大的区别在于数据的存取顺序：
  • 栈：遵循“后进先出”（Last In, First Out，简称 LIFO）的原则，最后进入栈的元素会首先被取出；
  • 队列：遵循“先进先出”（First In, First Out，简称 FIFO）的原则，最先进入队列的元素会最先被取出。

栈像是一个装东西的桶，最后放进去的东西会最先拿出来；队列像是排队买票，最先排队的人可以最先买到票离开。


图 1 栈和队列

图 1 演示了数据 {1, 2, 3, 4} 分别在栈和队列中的存储状态。

关于栈和队列各自的具体实现，前面章节已经做了详细的介绍，这里不再赘述。本节的重点是教大家用两个栈模拟实现一个队列，换句话说，两个栈存取数据时要遵循队列的“先进先出”原则。

两个栈实现队列

用两个栈实现队列，重点是用两个栈模拟出队列添加元素（入队）和删除元素（出队）的过程。

观察图 1 不难发现，数据 {1, 2, 3, 4} 入栈和入队的顺序是一样的，这意味着模拟数据入队的过程用一个栈就可以实现。

再次观察图 1，数据 {1, 2, 3, 4} 出栈和出队的顺序是不一样的。{1, 2, 3, 4} 出队的顺序是1 2 3 4，要想栈中的数据也按照此顺序出栈，那么每次出栈的就必须是位于栈底的元素。

想获取栈顶的元素，可以借助另一个栈实现。为了方便表述，这里假设数据存放在栈 A 中，栈 B 是一个空栈，实现思路是：将 A 中的元素依次出栈，每出栈一个元素，就将该元素存储到 B 中。当 A 中所有的元素全部转存到 B 里，此时 B 的栈顶元素就是原来 A 的栈底元素。下图演示了整个实现的过程：


图 2 借助另一个栈获取栈底元素

由此，两个栈（A 和 B）模拟队列的方案就有了：
  • 模拟入队：用栈 A 模拟入队操作。当有新的元素入队时，先将栈 B 的元素全部转移到栈 A，再将新元素入栈到 A 中；
  • 模拟出队：借助栈 B 模拟出队操作。执行出队操作时，先将栈 A 的元素全部转移到栈 B，再将 B 的栈顶元素出栈；
  • 返回处于队头的元素：所谓队头的元素，就是最先出队的元素。先将栈 A 的元素全部转移到栈 B，B 的栈顶元素就是队头元素；
  • 判断队列是否为空：如果 A 和 B 都为空，则表明队列是空的。

接下来，先实现栈的基本操作，C 语言代码如下： 
typedef struct
{
    int num[10];
    int top;  // 栈顶
}Stack;
// 初始化栈
void initStack(Stack* ps)
{
    memset(ps->num, 0, 10 * sizeof(int));
    ps->top = -1;
}
// 入栈
void stackPush(Stack* ps, int data)
{
    if (ps->top == 10) {
        printf("栈已满，入栈操作失败\n");
        return;
    }
    (ps->num)[++ps->top] = data;
}
// 出栈
void stackPop(Stack* ps)
{
    if (ps->top == -1) {
        printf("栈已空，出栈操作失败\n");
        return;
    }
    ps->top--;
}
// 获取栈顶元素
int stackTop(Stack* ps)
{
    if (ps->top == -1) {
        printf("栈已空，出栈操作失败\n");
        return;
    }
    return (ps->num)[ps->top];
}
// 检测栈是否为空
int stackEmpty(Stack* ps)
{
    if (ps->top == -1)
    {
        return 1;
    }
    else
    {
        return 0;
    }
}

在此基础上，用两个栈模拟队列，下面是对应的 C 语言实现代码： 
//队列的定义
typedef struct {
    Stack A;//入队列栈
    Stack B;//出队列栈
} Queue;
//队列的创建
void queueCreate(Queue* q)
{
    initStack(&(q->A));
    initStack(&(q->B));
}
//入队
void queuePush(Queue* q, int x) {
    //如果 A 为空，将 B 中的元素全部转移到 A
    //如果 A 不为空，此时 B 中不会存在任何元素
    if (stackEmpty(q->A))
    {
        while (!stackEmpty(q->B))
        {
            int top = stackTop(q->B);
            stackPush(&(q->A), top);
            stackPop(&(q->B));
        }
    }
    // 将 x 入栈到 A 中
    stackPush(&(q->A), x);
}
//出队
int queuePop(Queue* q) {
    //如果 B 为空，将 A 中的元素全部转移到 A
    //如果 B 不为空，此时 A 中不会存在任何元素
    if (stackEmpty(q->B))
    {
        while (!stackEmpty(q->A))
        {
            int top = stackTop(q->A);
            stackPush(&(q->B), top);
            stackPop(&(q->A));
        }
    }
    //从 B 中去除栈顶元素，它就是原来 A 的栈底元素
    int top = stackTop(q->B);
    if (top != -1) {
        stackPop(&(q->B));
    }
    return top;
}
//判断队列是否为空
bool queueEmpty(Queue* q) {
    //当A和B同时为空时，队列才为空
    return stackEmpty(q->A) && stackEmpty(q->B);
}
//返回队头的元素
int queuePeek(Queue* q) {
    //如果 B 为空，将 A 中的元素全部转移到 A
    //如果 B 不为空，此时 A 中不会存在任何元素
    if (stackEmpty(q->B))
    {
        while (!stackEmpty(q->A))
        {
            int top = stackTop(q->A);
            stackPush(&(q->B), top);
            stackPop(&(q->A));
        }
    }
    //获取队头的元素
    int top = stackTop(q->B);
    return top;
}

#include <string.h>

}
// 出栈
void stackPop(Stack* ps)
{
    if (ps->top == -1) {

        return;

// 获取栈顶元素
int stackTop(Stack ps)
{
    if (ps.top == -1) {

        return -1;
    }
    return (ps.num)[ps.top];
}

int stackEmpty(Stack ps)

    if (ps.top == -1)

    }

//队列的创建

}
//入队
void queuePush(Queue* q, int x) {

    return stackEmpty(q->A) && stackEmpty(q->B);
}
//返回队头的元素
int queuePeek(Queue* q) {

    //如果 B 不为空，此时 A 中不会存在任何元素

    {
        while (!stackEmpty(q->A))

            int top = stackTop(q->A);

int main() {
    Queue q;
    queueCreate(&q);
    queuePush(&q, 1);
    queuePush(&q, 2);
    queuePush(&q, 3);
